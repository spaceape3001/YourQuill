macro(mt_core name)
    add_executable(test_${name} ${name}.cpp)
    target_link_libraries(test_${name} PRIVATE yqutil ${TBB_LIBRARIES})
endmacro()
macro(coretest name)
    mt_core(${name})
    add_test(NAME ${name} COMMAND test_${name} ${ARGN})
endmacro()
